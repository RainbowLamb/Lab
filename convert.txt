#include "stdafx.h"
#include <iostream>

int main()
{
	using namespace std;

	float input, c, f, k;
	char scale;
	printf("Enter input and scale:\n");
	scanf_s("%f ", &input); // ввод температуры в виде числа с плавающей точкой
	scanf_s("%c", &scale);	// ввод шкалы в виде одного символа
	switch (scale) {
		case 'C':							// если введена температура в Цельсиях 
			f = input * 9 / 5 + 32;
			k = input + 273.15;
			printf("%.2f", input);
			printf(" C:\n");
			printf("%.2f", f);
			printf(" F\n");
			printf("%.2f", k);
			printf(" K\n");
			break;
		case 'F':							// если введена температура в Фаренгейтах 
			c = (input - 32) * 5 / 9;
			k = (input - 32) * 5 / 9 + 273, 15;
			printf("%.2f", input);
			printf(" F:\n");
			printf("%.2f", c);
			printf(" C\n");
			printf("%.2f", k);
			printf(" K\n");
			break;
		case 'K':							// если введена температура в Кельвинах
			c = input - 273.15;
			f = (input - 273.15) * 9 / 5 + 32;
			printf("%.2f", input);
			printf(" K:\n");
			printf("%.2f", c);
			printf(" C\n");
			printf("%.2f", f);
			printf(" F\n");
			break;
		default:						// если у вас кривые руки, раз вы не смогли ввести все правильно
			printf("Incorrect input.\n");
	}


    return 0;
}
